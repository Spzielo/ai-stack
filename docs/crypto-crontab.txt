# Crypto One-Glance - CRON Configuration
# Add these lines to your crontab (crontab -e)

# Update top 50 list daily at 7:00 AM
0 7 * * * docker exec python-runner python scripts/collect_top50.py >> /tmp/crypto-top50.log 2>&1

# Collect all crypto metrics daily at 8:10 AM
10 8 * * * docker exec python-runner python scripts/collect_crypto_metrics.py >> /tmp/crypto-metrics.log 2>&1

# Compute scores daily at 9:00 AM
0 9 * * * curl -X POST http://localhost:8000/crypto/compute/scores >> /tmp/crypto-scores.log 2>&1
